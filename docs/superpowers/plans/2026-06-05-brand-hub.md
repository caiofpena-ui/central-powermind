> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build Subsistema 1 (Brand Hub) do PowerMind Marketing Hub — repositório editável da marca com geração de briefings via Claude e copy via GPT.

**Architecture:** marketing.html SPA com 4 subabas (Brand Hub ativa), servido por server.py. brand.json centraliza dados da marca; briefings.json guarda histórico. Claude API gera briefings UGC/arte; OpenAI GPT gera 3 variações de copy.

**Tech Stack:** Python 3 HTTPServer, HTML/CSS/JS vanilla, Anthropic claude-3-5-haiku-20241022, OpenAI gpt-4o-mini, urllib.request (sem dependências externas), JSON para persistência.

```python
def chamar_claude_api(system_prompt, user_prompt, api_key):
    import urllib.request as urlreq
    payload = json.dumps({
        "model": "claude-3-5-haiku-20241022",
        "max_tokens": 1024,
        "system": system_prompt,
        "messages": [{"role": "user", "content": user_prompt}]
    }).encode('utf-8')
    req = urlreq.Request(
        'https://api.anthropic.com/v1/messages',
        data=payload,
        headers={
            'x-api-key': api_key,
            'anthropic-version': '2023-06-01',
            'Content-Type': 'application/json'
        }
    )
    with urlreq.urlopen(req, timeout=30) as resp:
        result = json.loads(resp.read())
    return result['content'][0]['text']

def chamar_openai_api(system_prompt, user_prompt, api_key):
    import urllib.request as urlreq
    payload = json.dumps({
        "model": "gpt-4o-mini",
        "messages": [
            {"role": "system", "content": system_prompt},
            {"role": "user", "content": user_prompt}
        ],
        "temperature": 0.8
    }).encode('utf-8')
    req = urlreq.Request(
        'https://api.openai.com/v1/chat/completions',
        data=payload,
        headers={
            'Authorization': f'Bearer {api_key}',
            'Content-Type': 'application/json'
        }
    )
    with urlreq.urlopen(req, timeout=30) as resp:
        result = json.loads(resp.read())
    return result['choices'][0]['message']['content']
```

// ---- NAVEGAÇÃO ----
function mostrarAba(aba) {
  const abas = ['brand-hub', 'criativos', 'calendario', 'publicacao'];
  abas.forEach(a => {
    const el = document.getElementById('aba-' + a);
    if (el) el.style.display = a === aba ? '' : 'none';
  });
  document.querySelectorAll('.nav-item').forEach(n => n.classList.remove('ativo'));
  const navItems = document.querySelectorAll('.nav-item');
  const idx = abas.indexOf(aba);
  if (navItems[idx]) navItems[idx].classList.add('ativo');
}

// ---- CARDS ----
function toggleCard(id) {
  const card = document.getElementById('card-' + id);
  if (card) card.classList.toggle('aberto');
}

// ---- LISTAS EDITÁVEIS ----
function renderLista(containerId, items) {
  const container = document.getElementById(containerId);
  if (!container) return;
  container.innerHTML = '';
  (items || []).forEach(item => {
    const div = document.createElement('div');
    div.className = 'lista-item';
    div.innerHTML = '<input type="text" value="' + escHtml(item) + '"><button class="btn-remover" onclick="removerItem(this)">×</button>';
    container.appendChild(div);
  });
}

function adicionarItem(containerId) {
  const container = document.getElementById(containerId);
  const div = document.createElement('div');
  div.className = 'lista-item';
  div.innerHTML = '<input type="text" placeholder="Novo item..."><button class="btn-remover" onclick="removerItem(this)">×</button>';
  container.appendChild(div);
  div.querySelector('input').focus();
}

function removerItem(btn) {
  btn.parentElement.remove();
}

function getLista(containerId) {
  const container = document.getElementById(containerId);
  if (!container) return [];
  return Array.from(container.querySelectorAll('input')).map(i => i.value.trim()).filter(v => v);
}

// ---- CORES ----
function renderCores(containerId, cores) {
  const container = document.getElementById(containerId);
  if (!container) return;
  container.innerHTML = '';
  (cores || []).forEach(cor => {
    const div = document.createElement('div');
    div.className = 'cor-item';

}

// ---- PILARES ----
function renderPilares(pilares) {
  const container = document.getElementById('lista_pilares');
  if (!container) return;
  container.innerHTML = '';
  (pilares || []).forEach((p, i) => {
    const div = document.createElement('div');
    div.className = 'pilar-item';
    div.innerHTML =
      '<div class="pilar-row">' +
        '<input type="text" placeholder="Nome do pilar" value="' + escHtml(p.nome) + '" class="pilar-nome">' +
        '<input type="number" min="0" max="100" value="' + p.frequencia_pct + '" class="pilar-pct pct" onchange="atualizarTotalPct()">' +
      '</div>' +
      '<input type="text" placeholder="Descrição do pilar" value="' + escHtml(p.descricao) + '" class="pilar-desc" style="width:100%;background:var(--dark2);border:1px solid var(--borda);color:var(--texto);padding:8px 10px;border-radius:6px;font-size:13px">';
    container.appendChild(div);
  });
  atualizarTotalPct();
}

function getPilares() {
  const items = document.querySelectorAll('#lista_pilares .pilar-item');
  return Array.from(items).map(item => ({
    nome: item.querySelector('.pilar-nome').value.trim(),
    descricao: item.querySelector('.pilar-desc').value.trim(),
    frequencia_pct: parseInt(item.querySelector('.pilar-pct').value) || 0
  })).filter(p => p.nome);
}

function atualizarTotalPct() {
  const total = getPilares().reduce((s, p) => s + p.frequencia_pct, 0);
  const el = document.getElementById('total_pct');
  if (el) {
    el.textContent = 'Total: ' + total + '%' + (total !== 100 ? ' ⚠️ (deve ser 100%)' : ' ✅');
    el.style.color = total === 100 ? 'var(--verde)' : 'var(--amarelo)';
  }
}

function popularFormulario(d) {
  const pos = d.posicionamento || {};
  if (document.getElementById('pos_proposta_valor')) document.getElementById('pos_proposta_valor').value = pos.proposta_valor || '';
  if (document.getElementById('pos_missao')) document.getElementById('pos_missao').value = pos.missao || '';
  if (document.getElementById('pos_categoria')) document.getElementById('pos_categoria').value = pos.categoria || '';
  renderLista('lista_diferenciais', pos.diferenciais || []);

  const per = d.persona || {};
  if (document.getElementById('per_nome')) document.getElementById('per_nome').value = per.nome || '';
  if (document.getElementById('per_idade')) document.getElementById('per_idade').value = per.idade || '';
  if (document.getElementById('per_ocupacao')) document.getElementById('per_ocupacao').value = per.ocupacao || '';
  renderLista('lista_dores', per.dores || []);
  renderLista('lista_desejos', per.desejos || []);
  renderLista('lista_objecoes', per.objecoes || []);

  const tom = d.tom_de_voz || {};
  renderLista('lista_adjetivos', tom.adjetivos || []);
  renderLista('lista_falar', tom.falar || []);
  renderLista('lista_nao_falar', tom.nao_falar || []);

  const fw = d.copy_framework || {};
  if (document.getElementById('copy_headline')) document.getElementById('copy_headline').value = fw.headline_padrao || '';
  if (document.getElementById('copy_estrutura')) document.getElementById('copy_estrutura').value = fw.estrutura_copy || '';
  renderLista('lista_ganchos', fw.ganchos_validados || []);
  renderLista('lista_ctas', fw.ctas_aprovados || []);

  const vis = d.identidade_visual || {};
  renderCores('cores_primarias', vis.cores_primarias || []);
  renderCores('cores_secundarias', vis.cores_secundarias || []);
  if (document.getElementById('vis_estilo')) document.getElementById('vis_estilo').value = vis.estilo || '';
  renderLista('lista_fontes', vis.fontes || []);
  renderLista('lista_elementos', vis.elementos_visuais || []);

  renderPilares(d.pilares_conteudo || []);

  // Popular pilares no select de geração
  const sel = document.getElementById('gen_pilar');
  if (sel && d.pilares_conteudo) {
    sel.innerHTML = d.pilares_conteudo.map(p => '<option value="' + escHtml(p.nome) + '">' + escHtml(p.nome) + '</option>').join('');
  }
}

function gerarBriefing() {
  const tipo = document.getElementById('gen_tipo').value;
  const pilar = document.getElementById('gen_pilar').value;
  const tema = document.getElementById('gen_tema').value.trim();
  const canal = document.getElementById('gen_canal').value;

  document.getElementById('geracao_loading').style.display = '';
  document.getElementById('resultado_area').style.display = 'none';
  document.getElementById('variacoes_area').style.display = 'none';
  document.getElementById('btn_claude').disabled = true;

    method: 'POST',
    headers: {'Content-Type': 'application/json'},
    body: JSON.stringify({tipo, pilar, tema_livre: tema, canal_destino: canal})
  })
    .then(r => r.json())

      document.getElementById('resultado_area').style.display = '';
      _briefingAtual = data;
      carregarHistorico();
      mostrarToast('✅ Briefing gerado e salvo no histórico!');
    })
    .catch(e => {
      document.getElementById('geracao_loading').style.display = 'none';
      document.getElementById('btn_claude').disabled = false;
      mostrarToast('Erro: ' + e.message, true);
    });
}

function renderHistorico(items) {
  const container = document.getElementById('tabela_historico');
  if (!items || items.length === 0) {
    container.innerHTML = '<div style="color:var(--texto2);font-size:13px;padding:20px 0">Nenhum briefing gerado ainda.</div>';
    return;
  }
  const tagClass = {ugc_roteiro: 'tag-ugc', briefing_arte: 'tag-arte', copy: 'tag-copy'};
  const tagLabel = {ugc_roteiro: 'Roteiro UGC', briefing_arte: 'Briefing Arte', copy: 'Copy GPT'};
  container.innerHTML = '<table>' +
    '<thead><tr><th>Data</th><th>Tipo</th><th>Pilar</th><th>Canal</th><th>Ações</th></tr></thead>' +
    '<tbody>' +
    items.map(b => '<tr>' +
      '<td>' + formatarData(b.criado_em) + '</td>' +
      '<td><span class="tag ' + (tagClass[b.tipo] || '') + '">' + (tagLabel[b.tipo] || b.tipo) + '</span></td>' +
      '<td>' + escHtml(b.pilar || '') + '</td>' +
      '<td>' + escHtml(b.canal_destino || '') + '</td>' +
      '<td>' +
        '<button class="btn-acao" onclick="verBriefing(' + JSON.stringify(b.briefing || '').replace(/</g,'&lt;') + ')">Ver</button>' +
        '<button class="btn-acao" onclick="copiarTexto(' + JSON.stringify(b.briefing || '') + ')">Copiar</button>' +
      '</td>' +
    '</tr>').join('') +
    '</tbody></table>';
}

function verBriefing(texto) {
  document.getElementById('resultado_texto').value = texto;
  document.getElementById('resultado_area').style.display = '';
  document.getElementById('resultado_texto').scrollIntoView({behavior:'smooth'});
}

// ---- UTILS ----
function formatarData(iso) {
  if (!iso) return '';
  const d = new Date(iso);
  return d.toLocaleDateString('pt-BR') + ' ' + d.toLocaleTimeString('pt-BR', {hour:'2-digit',minute:'2-digit'});
}

function escHtml(s) {
  if (!s) return '';
  return String(s).replace(/&/g,'&amp;').replace(/</g,'&lt;').replace(/>/g,'&gt;').replace(/"/g,'&quot;');
}

function copiarTexto(texto) {
  navigator.clipboard.writeText(texto).then(() => mostrarToast('✅ Copiado!')).catch(() => {});
}

function mostrarToast(msg, erro) {
  const t = document.getElementById('toast');
  t.textContent = msg;
  t.className = 'toast' + (erro ? ' erro' : '');
  t.style.display = '';
  clearTimeout(t._timer);
  t._timer = setTimeout(() => { t.style.display = 'none'; }, 3000);
}
